// 1. Summary Cards
$sql_summary = "SELECT 
                    SUM(duty_rm) as total_cukai,
                    COUNT(id) as total_kenderaan
                FROM vehicle_inventory";

                COALESCE(g.nama, 'Tidak Diketahui') as nama, 
                COUNT(v.id) as vehicle_count 
            FROM vehicle_inventory v 
            LEFT JOIN gbpekema g ON v.gbpekema_id = g.id 
            GROUP BY nama 
            ORDER BY vehicle_count DESC";
$result_pie = $conn->query($sql_pie);
$pie_labels = [];
$pie_data = [];
if ($result_pie) {
    while ($row = $result_pie->fetch_assoc()) {
        $pie_labels[] = $row['nama'];
        $pie_data[] = (int)$row['vehicle_count'];
    }
}
$pie_chart_data = json_encode(['labels' => $pie_labels, 'data' => $pie_data]);

$total_eksais = $tax_data['total_eksais'] ?? 0;

// 5. Monthly Collection Data
$current_year = date('Y');
$sql_line = "SELECT 
                MONTH(payment_date) as month, 
                SUM(duty_rm) as monthly_total
             FROM vehicle_inventory
             WHERE YEAR(payment_date) = ? AND duty_rm > 0
             GROUP BY MONTH(payment_date)
             ORDER BY month ASC";

$stmt_line = $conn->prepare($sql_line);
$stmt_line->bind_param("i", $current_year);
$stmt_line->execute();
$result_line = $stmt_line->get_result();

$monthly_data = array_fill(1, 12, 0);
while ($row = $result_line->fetch_assoc()) {
    $monthly_data[(int)$row['month']] = (float)$row['monthly_total'];
}
$stmt_line->close();

$line_chart_labels = ["Jan", "Feb", "Mac", "Apr", "Mei", "Jun", "Jul", "Ogo", "Sep", "Okt", "Nov", "Dis"];

// AI FEATURES: Historical data for last 6 months
$sql_historical = "SELECT 
                        DATE_FORMAT(payment_date, '%Y-%m') as month,
                        SUM(duty_rm) as total,
                        COUNT(id) as count
                    FROM vehicle_inventory
                    WHERE payment_date >= DATE_SUB(NOW(), INTERVAL 6 MONTH)
                    AND duty_rm > 0
                    GROUP BY DATE_FORMAT(payment_date, '%Y-%m')
                    ORDER BY month ASC";
$result_historical = $conn->query($sql_historical);
$historical_data = [];
while ($row = $result_historical->fetch_assoc()) {
    $historical_data[] = [
        'month' => $row['month'],

// AI: Company performance data
$sql_company_performance = "SELECT 
                                g.nama,
                                COUNT(v.id) as vehicle_count,
                                SUM(v.duty_rm) as total_tax,
                                AVG(v.duty_rm) as avg_tax
                            FROM vehicle_inventory v
                            JOIN gbpekema g ON v.gbpekema_id = g.id
                            WHERE v.duty_rm > 0
                            GROUP BY g.nama
                            ORDER BY total_tax DESC
                            LIMIT 10";
$result_company_performance = $conn->query($sql_company_performance);
$company_performance = [];
if ($result_company_performance) {
    while ($row = $result_company_performance->fetch_assoc()) {
        $company_performance[] = [
            'name' => $row['nama'],
            'count' => intval($row['vehicle_count']),
            'total' => floatval($row['total_tax']),
            'avg' => floatval($row['avg_tax'])
        ];
    }
}

// AI: Anomaly detection - High value vehicles
$sql_anomalies = "SELECT v.vehicle_model, v.duty_rm, v.payment_date
                  FROM vehicle_inventory v
                  WHERE v.duty_rm > (SELECT AVG(duty_rm) * 2 FROM vehicle_inventory WHERE duty_rm > 0)
                  AND v.duty_rm > 0
                  ORDER BY v.duty_rm DESC
                  LIMIT 5";
$result_anomalies = $conn->query($sql_anomalies);
$anomalies = [];

// Get average for comparison
$avg_result = $conn->query("SELECT AVG(duty_rm) as avg_duty FROM vehicle_inventory WHERE duty_rm > 0");
if ($avg_result) {
    $avg_row = $avg_result->fetch_assoc();
    $avg_duty = $avg_row['avg_duty'] ?? 1;
}

if ($result_anomalies) {
    while ($row = $result_anomalies->fetch_assoc()) {
        $row['avg_duty'] = $avg_duty;
        $anomalies[] = $row;
    }
}

        function linearRegression(data) {
            const n = data.length;
            if (n < 2) return { slope: 0, intercept: data[0] || 0 };
            
            let sumX = 0, sumY = 0, sumXY = 0, sumX2 = 0;
            data.forEach((val, i) => {
                sumX += i;
                sumY += val;
                sumXY += i * val;
                sumX2 += i * i;
            });
            
            const slope = (n * sumXY - sumX * sumY) / (n * sumX2 - sumX * sumX);
            const intercept = (sumY - slope * sumX) / n;
            
            return { slope, intercept };
        }
